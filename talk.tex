\begin{thm}[D.]
For all $n$, we have $n^2= n \cdot n$.
\end{thm}

\pf With massive loss of generality, let $n=1$. Then we have
	\[
	1=1^2= 1 \cdot 1= 1
	\]
Therefore by overwhelming hope, it must always be true. \qed
\end{frame}
